foo.rb
```ruby
class Foo
  def sum(a, b)
    a + b
  end
end
```

spec/foo_spec.rb
```ruby
require 'rspec'
require "./foo.rb"

describe Foo do
  it "Summa 2 and 3 do 5" do
    dog = Foo.new
    expect(dog.sum(2, 3)).to eq(5)
  end
end
```
